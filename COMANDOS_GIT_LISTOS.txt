# Comandos para subir tu código a GitHub
# Ejecuta estos comandos UNO POR UNO en la terminal

# 1. Inicializar Git (solo primera vez)
git init

# 2. Agregar todos los archivos
git add .

# 3. Hacer commit
git commit -m "Initial commit: Pipeline de scraping ANI con Airflow"

# 4. Conectar con tu repositorio de GitHub
git remote add origin https://github.com/ArtDeros/prueba-tecnica-ani-scraping.git

# 5. Cambiar a rama main
git branch -M main

# 6. Subir a GitHub
git push -u origin main
